126LIGHTMODBUS_WARN_UNUSED uint16_t modbusCRC(const uint8_t *data, uint16_t length)
127{
128 uint16_t crc = 0xFFFF;
129
130 for (uint16_t i = 0; i < length; i++)
131 {
132 crc ^= (uint16_t) data[i];
133 for (uint8_t j = 8; j != 0; j--)
134 {
135 if ((crc & 0x0001) != 0)
136 {
137 crc >>= 1;
138 crc ^= 0xA001;
139 }
140 else
141 crc >>= 1;
142 }
143 }
