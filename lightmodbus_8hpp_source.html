91static inline void throwErrorInfo(ModbusErrorInfo err)
92{
93 if (modbusGetGeneralError(err) != MODBUS_OK)
94 throw GeneralError(modbusGetGeneralError(err));
95 else if (modbusGetRequestError(err) != MODBUS_OK)
96 throw RequestError(modbusGetRequestError(err));
97 else if (modbusGetResponseError(err) != MODBUS_OK)
98 throw ResponseError(modbusGetResponseError(err));
